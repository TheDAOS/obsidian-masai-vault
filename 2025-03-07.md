function findNextSmallerOfNextGreater(arr) {
    const n = arr.length;
    const nextGreater = new Array(n).fill(-1);
    const nextSmaller = new Array(n).fill(-1);
    const stack = [];

    // Step 1: Find the next greater element for each element
    for (let i = 0; i < n; i++) {
        while (stack.length > 0 && arr[i] > arr[stack[stack.length - 1]]) {
            nextGreater[stack.pop()] = arr[i];
        }
        stack.push(i);
    }

    // Step 2: Find the next smaller element for each element
    for (let i = 0; i < n; i++) {
        if (nextGreater[i] !== -1) {
            while (stack.length > 0 && arr[i] < arr[stack[stack.length - 1]]) {
                nextSmaller[stack.pop()] = arr[i];
            }
            stack.push(i);
        }
    }

    // Step 3: Print the results
    for (let i = 0; i < n; i++) {
        if (nextGreater[i] !== -1) {
            console.log(nextSmaller[i] !== -1 ? nextSmaller[i] : -1);
        } else {
            console.log(-1);
        }
    }
}

// Example usage:
const arr = [5, 1, 6, 2, 5, 1];
findNextSmallerOfNextGreater(arr);
